  <script>

    const SHEET_ID = 'YOUR_SHEET_ID'; // Replace with your Google Sheet ID

    const API_KEY = 'YOUR_API_KEY';   // Replace with your Google Sheets API key

    const SHEET_NAME = 'products_details'; // Your sheet name


    const productList = document.getElementById('productList');

    const searchInput = document.getElementById('searchInput');

    const searchColumn = document.getElementById('searchColumn');


    const modal = document.getElementById('productModal');

    const modalImg = document.getElementById('modalImg');

    const modalDesc = document.getElementById('modalDesc');

    const modalClose = document.getElementById('modalClose');


    let allProducts = [];


    async function fetchProducts() {

      const url = `https://sheets.googleapis.com/v4/spreadsheets/${SHEET_ID}/values/${SHEET_NAME}?key=${API_KEY}`;

      try {

        const res = await fetch(url);

        if (!res.ok) throw new Error('Failed to fetch data');

        const data = await res.json();


        // data.values is array of arrays; first row = headers

        const headers = data.values[0];

        const rows = data.values.slice(1);

        allProducts = rows.map(row => {

          return headers.reduce((obj, header, idx) => {

            obj[header] = row[idx] || '';

            return obj;

          }, {});

        });

        displayProducts(allProducts);

      } catch (err) {

        productList.innerHTML = '<p>Error loading products. Please check API key and Sheet ID.</p>';

        console.error(err);

      }

    }


    // Display products in list

    function displayProducts(products) {

      if (products.length === 0) {

        productList.innerHTML = '<p>No products found.</p>';

        return;

      }

      productList.innerHTML = '';

      products.forEach(prod => {

        const card = document.createElement('div');

        card.className = 'product-card';


        card.dataset.image = prod.Image_url;

        card.dataset.description = prod.description;


        const img = document.createElement('img');

        img.className = 'product-img';

        img.src = prod.Image_url || '';

        img.alt = prod.product_title;

        img.onerror = () => { img.src = ''; img.alt = 'No image'; };

        card.appendChild(img);


        const info = document.createElement('div');

        info.className = 'product-info';


        const title = document.createElement('div');

        title.className = 'product-title';

        title.textContent = prod.product_title;

        info.appendChild(title);


        const price = document.createElement('div');

        price.className = 'product-price';

        price.textContent = (prod.price && prod.currency) ? `${prod.price} ${prod.currency}` : '';

        info.appendChild(price);


        card.appendChild(info);


        // Open modal on image click

        img.addEventListener('click', () => {

          modalImg.src = prod.Image_url || '';

          modalDesc.textContent = prod.description || 'No description available.';

          modal.style.display = 'block';

        });


        productList.appendChild(card);

      });

    }


    // Filter products based on search and column

    function filterProducts() {

      const query = searchInput.value.trim().toLowerCase();

      const column = searchColumn.value;


      if (!query) {

        displayProducts(allProducts);

        return;

      }


      const filtered = allProducts.filter(prod => {

        if (column === 'all') {

          return Object.values(prod).some(val =>

            (val || '').toLowerCase().includes(query)

          );

        } else {

          const val = prod[column] || '';

          return val.toLowerCase().includes(query);

        }

      });


      displayProducts(filtered);

    }


    searchInput.addEventListener('input', filterProducts);

    searchColumn.addEventListener('change', filterProducts);


    modalClose.onclick = () => {

      modal.style.display = 'none';

    };


    window.onclick = (event) => {

      if (event.target === modal) {

        modal.style.display = 'none';

      }

    };


    // Load products on page load

    fetchProducts();
